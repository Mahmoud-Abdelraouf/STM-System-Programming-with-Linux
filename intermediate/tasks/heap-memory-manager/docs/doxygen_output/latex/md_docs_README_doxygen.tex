To begin using Doxygen, install it on your system using the following command\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{sudo apt-\/get install -\/y doxygen}

\end{DoxyCode}

\begin{DoxyCode}{0}
\DoxyCodeLine{sudo apt-\/get install -\/y graphviz}

\end{DoxyCode}

\begin{DoxyEnumerate}
\item {\bfseries{Generate a Default Configuration File\+:}}

Run the following command to create a {\ttfamily Doxyfile} configuration\+:

\`{}\`{}\`{}bash doxygen -\/g \`{}\`{}\`{}
\item $\ast$$\ast$\+Edit the Generated Configuration File ({\ttfamily Doxyfile})\+:$\ast$$\ast$

Open the {\ttfamily Doxyfile} in your preferred text editor and make the following modifications\+:
\end{DoxyEnumerate}


\begin{DoxyCode}{0}
\DoxyCodeLine{\# Set project name}
\DoxyCodeLine{PROJECT\_NAME = "{}My\_Project"{}}
\DoxyCodeLine{}
\DoxyCodeLine{\# Optimize output for C}
\DoxyCodeLine{OPTIMIZE\_OUTPUT\_FOR\_C = YES}
\DoxyCodeLine{}
\DoxyCodeLine{\# Extract all documentation, including static and private members}
\DoxyCodeLine{EXTRACT\_ALL = YES}
\DoxyCodeLine{EXTRACT\_STATIC = YES}
\DoxyCodeLine{EXTRACT\_PRIVATE = YES}
\DoxyCodeLine{}
\DoxyCodeLine{\# Set input directory (Assuming source and header files are in src/ and include/)}
\DoxyCodeLine{INPUT = src include}
\DoxyCodeLine{}
\DoxyCodeLine{\# Recursively search for input files}
\DoxyCodeLine{RECURSIVE = YES}
\DoxyCodeLine{}
\DoxyCodeLine{\# Generate source browser and include code comments}
\DoxyCodeLine{SOURCE\_BROWSER = YES}
\DoxyCodeLine{INLINE\_SOURCES = YES}
\DoxyCodeLine{STRIP\_CODE\_COMMENTS = NO}
\DoxyCodeLine{}
\DoxyCodeLine{\# Enable Graphviz for generating diagrams}
\DoxyCodeLine{HAVE\_DOT = YES}
\DoxyCodeLine{DOT\_NUM\_THREADS = 4}
\DoxyCodeLine{}
\DoxyCodeLine{\# Generate call graphs and caller graphs}
\DoxyCodeLine{CALL\_GRAPH = YES}
\DoxyCodeLine{CALLER\_GRAPH = YES}
\DoxyCodeLine{}
\DoxyCodeLine{\# Set output directory}
\DoxyCodeLine{OUTPUT\_DIRECTORY = docs/doxygen\_output}
\DoxyCodeLine{}
\DoxyCodeLine{\# Enable HTML output and LaTeX for PDF generation}
\DoxyCodeLine{GENERATE\_HTML = YES}
\DoxyCodeLine{GENERATE\_LATEX = YES}
\DoxyCodeLine{GENERATE\_PDF = YES}

\end{DoxyCode}

\hypertarget{md_docs_README_doxygen_autotoc_md18}{}\doxysubsection{$<$strong$>$\+Enabling Diagrams and Charts$<$/strong$>$}\label{md_docs_README_doxygen_autotoc_md18}

\begin{DoxyCode}{0}

\DoxyCodeLine{CLASS\_DIAGRAMS = YES}

\DoxyCodeLine{INCLUDE\_GRAPH = YES}
\DoxyCodeLine{INCLUDED\_BY\_GRAPH = YES}
\DoxyCodeLine{GRAPHICAL\_HIERARCHY = YES}
\DoxyCodeLine{DOT\_IMAGE\_FORMAT = svg}

\end{DoxyCode}
